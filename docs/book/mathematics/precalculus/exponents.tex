\section{Exponents}

\subsection*{Laws of Exponents}
$$
a^{m} \cdot a^{n} = a^{mn}
$$

$$
\frac{a^{m}}{a^{n}} = a^{m-n}
$$

$$
\left( a^{m} \right)^{n} = a^{m \cdot n}
$$

$$
\left( ab \right)^{n} = a^{n} \cdot b^{n}
$$

$$
\left( \frac{a}{b} \right) ^{n} = \frac{a^{n}}{b^{n}}
$$

$$
\left( \frac{a}{b} \right)^{-n} = \left( \frac{b}{a} \right)^{n}
$$

$$
\frac{a^{-m}}{b^{-n}} = \frac{b^{n}}{a^{m}}
$$
