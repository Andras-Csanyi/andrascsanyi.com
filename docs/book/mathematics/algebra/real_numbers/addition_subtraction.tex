\subsection{Addition and Subtraction}

\begin{tabularx}{1\textwidth}{
    p{\dimexpr0.5\textwidth\relax}
    p{\dimexpr0.5\textwidth\relax}
}
\toprule
\multicolumn{2}{c}{\textbf{Rules of Addition and Subtraction}} \\
\midrule

\textbf{Property} & \textbf{Example}\\
\midrule

\makecell[l]{
    $(-1)a = -a$
} 
& 
\makecell[l]{
    $(-1)5 = -5$
} 
\\
\makecell[l]{
    $-(-a) = a$
} 
& 
\makecell[l]{
    $-(-5) = 5$
} 
\\
\makecell[l]{
    $(-a)b = a(-b) = -(ab)$
} 
& 
\makecell[l]{
    $(-3)5 = 5(-3) = -(5 \cdot 3)$
} 
\\
\makecell[l]{
    $(-a)(-b) = ab$
} 
& 
\makecell[l]{
    $(-3)(-5) = 5 \cdot 3$
} 
\\
\makecell[l]{
    $-(a+b) = -a-b$
} 
& 
\makecell[l]{
    $-(3+5) = -5-3$
} 
\\
\makecell[l]{
    $-(a-b) = b-a = -a + b$
} 
& 
\makecell[l]{
    $-(3-5) = 5-3 = -3+5$
} 
\\

\bottomrule
\end{tabularx}
